  <script>

    function animateScore(who) {

      const status = document.getElementById('status');

      status.style.transform = 'scale(1.05)';

      if (who === 'player') {

        status.style.backgroundColor = '#228822';

      } else if (who === 'computer') {

        status.style.backgroundColor = '#aa2222';

      }

      setTimeout(() => {

        updateStatus();

        status.style.transform = 'scale(1)';

      }, 800);

    }


    const insults = [

  ["Deine Mutter ist so fett, sie hat ihre eigene Postleitzahl!", "Dann bekommt sie wenigstens mehr Fanpost als du!"],

  ["Deine Mutter ist so fett, wenn sie ins Meer springt, steigt der Meeresspiegel!", "Und trotzdem macht sie mehr Wellen als du!"],

  ["Deine Mutter ist so fett, sie braucht ein GPS, um sich umzudrehen!", "Und du brauchst eins, um einen Witz zu finden!"],

  ["Deine Mutter ist so fett, dass selbst Google Maps sie nicht ganz anzeigen kann!", "Immerhin bin ich leicht zu finden!"],

  ["Deine Mutter ist so fett, sie verursacht eine Sonnenfinsternis, wenn sie spazieren geht!", "Dann bring lieber deine Sonnenbrille mit!"],

  ["Deine Mutter ist so fett, sie isst beim All-you-can-eat das Gebäude mit!", "Und du kriegst nicht mal eine Einladung."],

  ["Deine Mutter ist so fett, sie hat ein eigenes Gravitationsfeld!", "Deshalb kreist du wohl ständig um sie herum!"],

  ["Deine Mutter ist so fett, sie verursacht Stau beim Rückwärtsgehen!", "Dann bleib du lieber stehen!"],

  ["Deine Mutter ist so fett, sie benutzt ein Sofa als Flip-Flop!", "Und trotzdem tritt sie dich mit Stil."],

  ["Deine Mutter ist so fett, sie macht dem Walross Konkurrenz!", "Das Walross hat sich schon bedankt!"],

  ["Deine Mutter ist so fett, wenn sie springt, denkt man, es sei ein Erdbeben!", "Dann wackel nicht so bei jeder Niederlage!"],

  ["Deine Mutter ist so fett, sie kann sich nicht im Spiegel sehen – es gibt keinen, der groß genug ist!", "Dein Ego passt auch in keinen Spiegel."],

  ["Deine Mutter ist so fett, sie hat ihren eigenen Parkplatz im Buffet-Restaurant!", "Immerhin braucht sie keinen für dich."],

  ["Deine Mutter ist so fett, sie wurde für eine Insel gehalten!", "Dann brauchst du wohl eine Schatzkarte!"],

  ["Deine Mutter ist so fett, sie braucht zwei Flugtickets – für einen Sitz!", "Und du fliegst trotzdem auf sie!"],

  ["Deine Mutter ist so fett, sie schmilzt in der Sonne nicht – sie blockiert sie!", "Zum Glück blendest du niemanden."],

  ["Deine Mutter ist so fett, sie hat beim Limbo das Tor eingerissen!", "Dann bleib lieber aufrecht!"],

  ["Deine Mutter ist so fett, sie wird in Megatonnen gewogen!", "Und du in Gramm Mut."],

  ["Deine Mutter ist so fett, wenn sie lacht, bebt der Planet!", "Dann bring sie lieber nicht zum Weinen."],

  ["Deine Mutter ist so fett, ihr Schatten hat einen eigenen Schatten!", "Dann bleib lieber im Licht."],

];


    let playerPoints = 0;

    let computerPoints = 0;

    let playerTurn = true;

    let currentInsultIndex = null;

    let awaitingPlayerResponse = false;

    let round = 1;


    const log = document.getElementById('gameLog');

    const buttons = document.getElementById('insultButtons');

    const status = document.getElementById('status');


    function updateStatus() {

      status.innerHTML = `${playerTurn ? '🧑 Du bist dran!' : '🤖 Der Computer ist dran!'}<br>Runde: ${round} | Du: ${playerPoints} Punkte | Computer: ${computerPoints} Punkte`;

    }


    function showMessage(msg) {

      log.innerHTML += `<p>${msg}</p>`;

      log.scrollTop = log.scrollHeight;

    }


    function checkWin() {

      if (playerPoints >= 3) {

        showMessage("<strong>Du hast gewonnen!</strong>");

        buttons.innerHTML = "";

        return true;

      }

      if (computerPoints >= 3) {

        showMessage("<strong>Der Computer hat gewonnen!</strong>");

        buttons.innerHTML = "";

        return true;

      }

      return false;

    }


    function nextRound() {

      if (playerPoints < 3 && computerPoints < 3) {

        round++;

        updateStatus();

        if (playerTurn) {

          showMessage("Du bist dran. Wähle eine Beleidigung:");

          showInsultOptions();

        } else {

          showMessage("Computer ist dran mit einer Beleidigung...");

          setTimeout(computerInsults, 1000);

        }

      }

    }


    function computerResponds() {

      const correct = Math.random() < 0.6;

      let response;

      if (correct) {

        response = insults[currentInsultIndex][1];

        showMessage(`<em>Computer antwortet:</em> ${response}`);

        computerPoints++;

        animateScore('computer');

        showMessage("Computer bekommt einen Punkt!");

        playerTurn = false;

      } else {

        response = "Ähm... was meinst du damit?";

        showMessage(`<em>Computer antwortet falsch:</em> ${response}`);

        showMessage("Falsche Antwort! Du bekommst einen Punkt.");

        playerPoints++;

        animateScore('player');

        playerTurn = true;

      }

      updateStatus();

      if (!checkWin()) setTimeout(nextRound, 1500);

    }


    function computerInsults() {

      currentInsultIndex = Math.floor(Math.random() * insults.length);

      const insult = insults[currentInsultIndex][0];

      showMessage(`<em>Computer sagt:</em> ${insult}`);

      showMessage("Wähle eine Antwort:");

      showResponseOptions();

    }


    function showResponseOptions() {

      buttons.className = "response-mode";

      buttons.innerHTML = "";

      awaitingPlayerResponse = true;

      insults.forEach((pair, index) => {

        const btn = document.createElement("button");

        btn.textContent = pair[1];

        btn.onclick = () => {

          showMessage(`<strong>Du antwortest:</strong> ${pair[1]}`);

          if (index === currentInsultIndex) {

            showMessage("Du konterst erfolgreich!");

            playerPoints++;

            animateScore('player');

            playerTurn = true;

          } else {

            showMessage("Falsche Antwort!");

            computerPoints++;

            animateScore('computer');

            playerTurn = false;

          }

          awaitingPlayerResponse = false;

          updateStatus();

          if (!checkWin()) setTimeout(nextRound, 1500);

        };

        buttons.appendChild(btn);

      });

    }


    function showInsultOptions() {

      if (awaitingPlayerResponse) return;

      buttons.className = "insult-mode";

      buttons.innerHTML = "";

      insults.forEach((pair, index) => {

        const btn = document.createElement("button");

        btn.textContent = pair[0];

        btn.onclick = () => {

          currentInsultIndex = index;

          showMessage(`<strong>Du sagst:</strong> ${pair[0]}`);

          setTimeout(() => computerResponds(), 1000);

        };

        buttons.appendChild(btn);

      });

    }


    updateStatus();

    showMessage("Du beginnst. Wähle eine Beleidigung:");

    showInsultOptions();
